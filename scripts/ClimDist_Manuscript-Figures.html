sem <- function(x, na.rm=FALSE) {           #for caclulating standard error
  sd(x,na.rm=na.rm)/sqrt(length(na.omit(x)))
} 

arcsin_transform <- function(x) { #for arcsine transforming probabilities 
  asin(sqrt(x))
}
